Use los tabs para segmentar y organizar el contenido utilizando un mismo bloque de espacio visible.

Creamos el componente a traves de la terminal, con el comando ng generate component. 

ng generate component shared/components/tab --skip-tests

Una vez creado el componente, agregamos la siguiente estructura html 

<mat-tab-group class="tab" animationDuration="0ms">
    <mat-tab label="Tab Uno">
        <ng-template matTabContent>
            <p>Contenido Uno</p>
            { { getContentLoaded(1) } }
        </ng-template>
    </mat-tab>
    <mat-tab label="Tab Dos">
        <ng-template matTabContent>
            <p>Contenido Dos</p>
            { { getContentLoaded(2) } }
        </ng-template>
    </mat-tab>
</mat-tab-group>

Hay que tener en cuenta que este componente debe ir dentro de las estructuras de filas y columnnas de Bootstrap. 

<div class="row" >
    <div class="col-md-5">
        <tab></tab>
    </div>
</div>

En el archivo tab.component.ts añadir la siguiente programación. 

import { Component, OnInit } from '@angular/core';
                    
@Component({
    selector: 'tab',
    templateUrl: './tab.component.html',
    styleUrls: ['./tab.component.scss']
})

export class TabComponent implements OnInit {

    tabLoadContent: string[] = [];

    constructor() { }

    ngOnInit(): void {
    };

    getContentLoaded(index: number) {
        if ( !this.tabLoadContent[index] ) {
            this.tabLoadContent[index] = '';
        }

        return this.tabLoadContent[index];
    }
}

Tab Verticales

Podemos usar tabs en modo vertical solamente con agregar la clase tab-vertical 

<mat-tab-group class="tab tab-vertical" animationDuration="0ms">
    <mat-tab label="Tab Uno">
        <ng-template matTabContent>
            <p>Contenido Uno</p>
            { { getContentLoaded(1) } }
        </ng-template>
    </mat-tab>
    <mat-tab label="Tab Dos">
        <ng-template matTabContent>
            <p>Contenido Dos</p>
            { { getContentLoaded(2) } }
        </ng-template>
    </mat-tab>
    <mat-tab label="Tab Tres">
        <ng-template matTabContent>
            <p>Contenido Tres</p>
            { { getContentLoaded(3) } }
        </ng-template>
    </mat-tab>
</mat-tab-group>
